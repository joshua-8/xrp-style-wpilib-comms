218 // recall data from list of received messages
219 template <typename T>
220 bool getData(T& data, const uint8_t id)
221 {
222 for (MessageType* msg : receivedMessages) {
223 if (msg->getTag() == TYPE_TO_TAG_VAL(T) && (msg->hasId() == false || msg->getId() == id)) {
224 // Check if we've found a message of the correct type and with the correct ID
225 data = *static_cast<T*>(msg->getData());
226 return true; // Data found
227 }
228 }
229 return false; // No data found
230 }
231
232 // save data into list of messages to send
233 template <typename T>
234 bool sendData(T data, bool checkUniqueness)
235 {
236 MessageType* message = nullptr;
237 if (checkUniqueness) {
238 // search sent messages for a message with tag and id
239 for (MessageType* msg : sentMessages) {
240 if (msg->getTag() == TYPE_TO_TAG_VAL(T) && (msg->hasId() == false || msg->getId() == data.id)) {
241 // Found a message of the correct type and with the correct ID
242 // we'll overwrite it
243 message = msg;
244 break;
245 }
246 }
247 }
248 if (message == nullptr) {
249 // No existing message found, create a new one
250 message = MessageTypeFactory::createMessageType(TYPE_TO_TAG_VAL(T));
251 }
252 if (message != nullptr) {
253 message->setData(&data);
254 sentMessages.push_back(message);
255 return true; // Data successfully added to the list
256 }
257 return false;
258 }
